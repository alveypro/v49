def calculate_risk_penalty(indicators):
    """系统性风险评估，扣分机制"""
    penalty = 0
    
    # 高位风险
    if (indicators['gain_60d'] > 0.50 and 
        indicators['recent_decline']):
        penalty += 10
        
    # 波动风险
    volatility = indicators['std_20d']
    if volatility > 0.15:
        penalty += 8
    elif volatility > 0.10:
        penalty += 5
        
    # 暴跌风险
    if indicators['limit_down_count_60d'] >= 2:
        penalty += 8
    elif indicators['limit_down_count_60d'] >= 1:
        penalty += 5
        
    # 技术破位风险
    if indicators['below_all_ma']:

    # 流动性风险
    if indicators['turnover_rate'] < 0.005:
        penalty += 5
    elif indicators['turnover_rate'] < 0.01:
        penalty += 3
        
    return min(41, penalty)  # 最高41分
