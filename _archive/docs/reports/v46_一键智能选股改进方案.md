### 改进1：优化阈值策略

```python
# 原代码（第949行）：
stocks = vp_analyzer.select_current_stocks_complete(df, min_strength=70)

# 改进代码：
# 智能阈值递降策略
thresholds = [65, 60, 55]  # 从65开始，更合理
result_stocks = None

for threshold in thresholds:
    stocks = vp_analyzer.select_current_stocks_complete(df, min_strength=threshold)
    
    if not stocks.empty:
        # 板块过滤
        hot_stocks = stocks[stocks['industry'].isin(hot_sectors)]
        
        if len(hot_stocks) >= 15:
            result_stocks = hot_stocks
            st.info(f"✅ 使用阈值{threshold}分，找到{len(hot_stocks)}只符合条件的股票")
            break
        elif len(hot_stocks) >= 5:
            # 5-14只，降级提醒
            result_stocks = hot_stocks
            st.warning(f"⚠️ 阈值{threshold}分找到{len(hot_stocks)}只（建议15+），自动尝试更低阈值...")
            continue
    
    if threshold == thresholds[-1] and result_stocks is None:
        # 最后一次尝试，使用全部结果
        result_stocks = stocks
        st.warning(f"⚠️ 即使使用{threshold}分阈值，热门板块股票也不足，展示全部{len(stocks)}只")

stocks = result_stocks
```

### 改进2：优化板块匹配逻辑

```python
# 原代码（第959行）：
stocks['is_hot_sector'] = stocks['industry'].isin(hot_sectors)

# 改进代码：
def fuzzy_match_sector(industry, hot_sectors):
    """模糊匹配板块"""
    if pd.isna(industry):
        return False
    
    industry = str(industry).strip()
    
    # 1. 精确匹配
    if industry in hot_sectors:
        return True
    
    # 2. 关键词匹配
    for hot_sector in hot_sectors:
        # 提取关键词（去除"及"、"与"等连词）
        sector_keywords = hot_sector.replace('及', ' ').replace('与', ' ').split()
        
        for keyword in sector_keywords:
            if len(keyword) >= 2 and keyword in industry:
                return True
            if len(industry) >= 2 and industry in hot_sector:
                return True
    
    return False

# 应用模糊匹配
stocks['is_hot_sector'] = stocks['industry'].apply(
    lambda x: fuzzy_match_sector(x, hot_sectors)
)

# 统计匹配情况
matched_count = stocks['is_hot_sector'].sum()
total_count = len(stocks)
st.info(f"📊 板块匹配：{matched_count}/{total_count} ({matched_count/total_count*100:.1f}%)")
```

### 改进4：显示详细的过滤信息

```python
# 在板块筛选后添加详细信息显示
if hot_sectors:
    st.divider()
    st.subheader("📊 板块筛选详情")
    
    # 统计每个热门板块的股票数量
    sector_stats = {}
    for sector in hot_sectors:
        count = stocks[stocks['industry'].apply(
            lambda x: fuzzy_match_sector(x, [sector])
        )].shape[0]
        sector_stats[sector] = count
    
    # 展示统计
    col1, col2 = st.columns([1, 2])
    with col1:
        st.markdown("**热门板块股票分布**")
        for sector, count in sorted(sector_stats.items(), key=lambda x: x[1], reverse=True):
            if count > 0:
                st.markdown(f"- {sector}: **{count}只** ✅")
            else:
                st.markdown(f"- {sector}: 0只 ⚠️")
    
    with col2:
        # 显示被过滤的高分股票
        filtered_stocks = stocks[~stocks['is_hot_sector']].head(10)
        if not filtered_stocks.empty:
            st.markdown("**被过滤的高分股票（非热门板块）**")
            st.dataframe(
                filtered_stocks[['name', 'industry', 'buy_value', 'signal_strength']],
                hide_index=True
            )
```
