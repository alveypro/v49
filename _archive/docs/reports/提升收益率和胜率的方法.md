score = (
    底部放量(30分) +
    量价背离(25分) +
    均线多头(20分) +
    动量强劲(15分)

# 改进评分（增加行业强度）
score = (
    底部放量(25分) +
    量价背离(20分) +
    均线多头(15分) +
    动量强劲(15分) +
    行业强度(25分)  # 新增！
) / 100 * 100

# 行业强度评分
if 股票所在行业 in 萌芽期板块:
    行业强度 = 25分
elif 股票所在行业 in 启动期板块:
    行业强度 = 20分
elif 股票所在行业 in 爆发期板块:
    行业强度 = 15分

    行业强度 = 5分
