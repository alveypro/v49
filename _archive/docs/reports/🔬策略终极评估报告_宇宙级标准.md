class RiskManagementSystem:
    """顶级风险管理系统"""
    
    def __init__(self):
        # 1. 投资组合层面
        self.max_single_position = 0.05  # 单只不超过5%
        self.max_sector_exposure = 0.20  # 单个行业不超过20%
        self.target_num_stocks = 15-20   # 持仓15-20只
        
        # 2. 风险预算
        self.max_daily_loss = 0.02       # 单日最大亏损2%
        self.max_drawdown = 0.15         # 最大回撤15%
        
        # 3. 动态仓位
        self.base_position = 0.05        # 基础仓位5%
        self.max_position = 0.10         # 最大仓位10%
        
        # 4. 止损策略
        self.stop_loss_pct = 0.07        # 止损7%
        self.trailing_stop = 0.05        # 移动止损5%
    
    def calculate_position_size(self, signal_strength, volatility):
        """
        根据信号强度和波动率计算仓位
        
        凯利公式：f = (p*b - q) / b
        其中：
        p = 胜率
        b = 盈亏比
        q = 1 - p
        """
        p = self.win_rate
        b = self.profit_loss_ratio
        q = 1 - p
        
        # 凯利仓位
        kelly = (p * b - q) / b
        
        # 保守起见，使用半凯利或1/4凯利
        conservative_kelly = kelly * 0.25
        
        # 考虑信号强度
        signal_factor = signal_strength / 100
        
        # 考虑波动率（波动越大，仓位越小）
        vol_factor = 1 / (1 + volatility)
        
        # 最终仓位
        position = conservative_kelly * signal_factor * vol_factor
        
        # 限制在合理范围
        return max(self.base_position, 
                  min(position, self.max_position))

class PositionManager:
    """仓位管理器"""
    
    def __init__(self, total_capital):
        self.total_capital = total_capital
        self.max_single_position = 0.05  # 5%
        self.max_total_position = 0.80   # 80%
        self.positions = {}
    
    def calculate_position_size(self, stock, signal_score):
        """
        根据信号强度计算仓位
        
        凯利公式的保守版本
        """
        # 基础仓位
        base_size = self.total_capital * 0.03  # 3%
        
        # 信号调整
        if signal_score >= 80:
            multiplier = 1.5
        elif signal_score >= 70:
            multiplier = 1.2
        elif signal_score >= 60:
            multiplier = 1.0
        else:
            multiplier = 0.8
        
        # 最终仓位
        position_size = base_size * multiplier
        
        # 限制
        max_allowed = self.total_capital * self.max_single_position
        return min(position_size, max_allowed)

class StopLossManager:
    """止损管理器"""
    
    def __init__(self):
        self.stop_loss_pct = 0.07  # 固定止损7%
        self.trailing_stop_pct = 0.05  # 移动止损5%
    
    def update_stop_loss(self, entry_price, current_price, highest_price):
        """
        更新止损价格
        
        策略：
        1. 初始：固定止损7%
        2. 盈利后：移动止损保护利润
        """
        # 初始止损
        initial_stop = entry_price * (1 - self.stop_loss_pct)
        
        # 如果盈利，使用移动止损
        if current_price > entry_price * 1.02:  # 盈利>2%
            trailing_stop = highest_price * (1 - self.trailing_stop_pct)
            # 取较高者
            return max(initial_stop, trailing_stop)
        
        return initial_stop

class PortfolioManager:

        """
        从候选池中选择投资组合
        
        优化目标：
        1. 最大化收益
        2. 最小化风险（分散）
        3. 行业平衡

        sector_exposure = {}

            current_exposure = sector_exposure.get(sector, 0)
            
            if current_exposure < self.max_sector_exposure:
                portfolio.append(stock)
                sector_exposure[sector] = current_exposure + stock['weight']
                
                # 达到目标数量
                if len(portfolio) >= self.target_num_stocks:
                    break
        
        return portfolio
```

**预期效果**：
- 夏普比率：0.07 → 0.5-1.0（7-14倍提升！）
- 最大回撤：30% → 15-20%
- 稳定性：大幅提升

### 方案B：多策略组合（次优先）⭐⭐⭐⭐

#### 为什么组合多个策略？

```
单一策略的问题：

- 提高夏普比率
- 适应不同市场环境

数学原理：
相关系数0.3的两个策略组合
组合波动率 = √(σ₁² + σ₂² + 2ρσ₁σ₂)
             ≈ 0.7 × 单个策略波动率

结论：波动降低30%，夏普提升43%！

```python
class MultiStrategySystem:
    """多策略组合系统"""
    
    def __init__(self):
        # 三个策略
        self.strategies = {
            'v4.0': {
                'weight': 0.40,  # 40%资金
                '适用环境': '震荡市',
                '预期收益': '15-20%',
                '预期夏普': 0.5
            },

                '适用环境': '趋势市',
                '预期收益': '20-25%',
                '预期夏普': 0.7
            },
            'v6.0': {
                'weight': 0.25,  # 25%资金
                '适用环境': '强势市',
                '预期收益': '25-30%',
                '预期夏普': 1.0
            }
        }
    
    def allocate_capital(self, market_environment):
        """
        根据市场环境动态调整权重
        
        市场识别：
        - 震荡市：大盘波动<1%/天
        - 趋势市：大盘连续5天>2%
        - 强势市：大盘连续10天>3%
        """
        if market_environment == '震荡市':
            return {'v4.0': 0.50, 'v5.0': 0.30, 'v6.0': 0.20}
        elif market_environment == '趋势市':
            return {'v4.0': 0.30, 'v5.0': 0.50, 'v6.0': 0.20}
        elif market_environment == '强势市':
            return {'v4.0': 0.20, 'v5.0': 0.30, 'v6.0': 0.50}
        else:
            return {'v4.0': 0.40, 'v5.0': 0.35, 'v6.0': 0.25}
    
    def combine_signals(self, signals_v4, signals_v5, signals_v6):
        """
        组合三个策略的信号
        
        方法：
        1. 投票法（至少2个策略看好）
        2. 加权法（根据市场环境加权）
        3. 集成法（机器学习组合）
        """
        combined_signals = []
        
        # 所有候选股票
        all_stocks = set()
        all_stocks.update([s['ts_code'] for s in signals_v4])
        all_stocks.update([s['ts_code'] for s in signals_v5])
        all_stocks.update([s['ts_code'] for s in signals_v6])
        
        for ts_code in all_stocks:
            votes = 0
            weighted_score = 0
            
            # v4.0投票
            if ts_code in [s['ts_code'] for s in signals_v4]:
                votes += 1
                score_v4 = [s['score'] for s in signals_v4 if s['ts_code'] == ts_code][0]
                weighted_score += score_v4 * 0.4
            
            # v5.0投票
            if ts_code in [s['ts_code'] for s in signals_v5]:
                votes += 1
                score_v5 = [s['score'] for s in signals_v5 if s['ts_code'] == ts_code][0]
                weighted_score += score_v5 * 0.35
            
            # v6.0投票
            if ts_code in [s['ts_code'] for s in signals_v6]:
                votes += 1
                score_v6 = [s['score'] for s in signals_v6 if s['ts_code'] == ts_code][0]
                weighted_score += score_v6 * 0.25
            
            # 至少2个策略看好
            if votes >= 2:
                combined_signals.append({
                    'ts_code': ts_code,
                    'votes': votes,
                    'weighted_score': weighted_score,
                    'confidence': 'HIGH' if votes == 3 else 'MEDIUM'
                })
        
        return sorted(combined_signals, 
                     key=lambda x: x['weighted_score'], 
                     reverse=True)
```

from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import TimeSeriesSplit
import pandas as pd

class MLEnhancedStrategy:
    """机器学习增强策略"""
    
    def __init__(self):
        self.model = RandomForestClassifier(
            n_estimators=500,
            max_depth=10,
            min_samples_split=100,
            random_state=42
        )
    
    def prepare_data(self, historical_data):
        """
        准备训练数据
        
        特征：
        - v4.0的8维得分
        - v5.0的8维得分
        - v6.0的7维得分
        - 市场环境特征
        - 技术指标
        
        标签：
        - 未来5日收益>3%: 1
        - 否则: 0
        """
        features = []
        labels = []
        
        for record in historical_data:
            # 提取特征
            feature_vector = [
                record['v4_lurking_value'],
                record['v4_bottom_feature'],
                record['v4_volume_price'],
                # ... 其他特征
                record['market_environment'],
                record['sector_strength'],
            ]
            
            # 提取标签
            future_return = record['future_5d_return']
            label = 1 if future_return > 3 else 0
            
            features.append(feature_vector)
            labels.append(label)
        
        return pd.DataFrame(features), pd.Series(labels)
    
    def train(self, X, y):
        """
        训练模型
        
        使用时间序列交叉验证
        - 避免未来数据泄露
        - 更真实的表现评估
        """
        tscv = TimeSeriesSplit(n_splits=5)
        
        scores = []
        for train_idx, test_idx in tscv.split(X):
            X_train, X_test = X.iloc[train_idx], X.iloc[test_idx]
            y_train, y_test = y.iloc[train_idx], y.iloc[test_idx]
            
            self.model.fit(X_train, y_train)
            score = self.model.score(X_test, y_test)
            scores.append(score)
        
        print(f"交叉验证平均准确率: {np.mean(scores):.2%}")
        
        # 用全部数据训练最终模型
        self.model.fit(X, y)
    
    def predict(self, stock_features):
        """
        预测
        
        返回：
        - 预测概率
        - 特征重要性
        """
        proba = self.model.predict_proba([stock_features])[0]
        
        return {
            'buy_probability': proba[1],
            'confidence': 'HIGH' if proba[1] > 0.7 else 'MEDIUM' if proba[1] > 0.6 else 'LOW'
        }
    
    def get_feature_importance(self):
        """
        获取特征重要性
        
        了解哪些特征最重要
        """
        importance = self.model.feature_importances_
        return sorted(zip(self.feature_names, importance), 
                     key=lambda x: x[1], 
                     reverse=True)
