```python
def scan_all_sectors_COMPLETE(self, days: int = 60) -> Dict:

    all_data = self._get_all_sectors_data(days)
    sectors = all_data['industry'].unique()
    
    results = {
        'emerging': [],
        'launching': [],
        'exploding': [],
        'declining': []
    }
    
    # 2. 并行分析所有板块
    with ThreadPoolExecutor(max_workers=10) as executor:
        futures = {}
        for sector in sectors:
            sector_data = all_data[all_data['industry'] == sector]
            future = executor.submit(
                self._analyze_sector_complete,
                sector, sector_data
            )
            futures[future] = sector
        
        # 3. 收集结果
        for future in as_completed(futures):
            result = future.result()
            if result:
                stage = result['stage']
                if stage == '萌芽期':
                    results['emerging'].append(result)
                elif stage == '启动期':
                    results['launching'].append(result)
                # ...
    
    return results
```

```python
def calculate_correlation_matrix(self, sectors: List[str]) -> pd.DataFrame:
    """计算板块相关性矩阵"""
    
    # 1. 获取所有板块收益率数据
    returns = {}
    for sector in sectors:
        sector_data = self._get_sector_data(sector, days=60)
        daily_return = sector_data.groupby('trade_date')['pct_chg'].mean()
        returns[sector] = daily_return
    
    # 2. 计算相关系数
    df = pd.DataFrame(returns)
    corr_matrix = df.corr()
    
    return corr_matrix
```
