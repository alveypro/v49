# 审计日志系统
import logging
from datetime import datetime
import json

class AuditLogger:
    """审计日志器（符合SOC 2标准）"""
    
    def __init__(self, log_file='audit.log'):
        self.logger = logging.getLogger('audit')
        handler = logging.FileHandler(log_file)
        formatter = logging.Formatter(
            '%(asctime)s - %(message)s',
            datefmt='%Y-%m-%d %H:%M:%S'
        )
        handler.setFormatter(formatter)
        self.logger.addHandler(handler)
        self.logger.setLevel(logging.INFO)
    
    def log_action(self, user: str, action: str, details: dict):
        """记录用户操作"""
        log_entry = {
            'timestamp': datetime.now().isoformat(),
            'user': user,
            'action': action,
            'details': details,
            'ip': self._get_client_ip(),
            'user_agent': self._get_user_agent()
        }
        self.logger.info(json.dumps(log_entry))
    
    def _get_client_ip(self) -> str:
        """获取客户端IP"""
        # 从Streamlit session获取
        return st.session_state.get('client_ip', 'unknown')
    
    def _get_user_agent(self) -> str:
        """获取用户代理"""
        return st.session_state.get('user_agent', 'unknown')

# 使用审计日志
audit = AuditLogger()

def scan_stocks_with_audit(user, strategy, threshold):
    """带审计的股票扫描"""
    audit.log_action(
        user=user,
        action='SCAN_STOCKS',
        details={
            'strategy': strategy,
            'threshold': threshold,
            'timestamp': datetime.now().isoformat()
        }
    )
    
    # 执行扫描
    results = scan_stocks(strategy, threshold)
    
    audit.log_action(
        user=user,
        action='SCAN_COMPLETE',
        details={
            'results_count': len(results),
            'top_score': max([r['score'] for r in results]) if results else 0
        }
    )
    
    return results

##### 4. 缺少速率限制（A04:2021）⚠️

**问题**：可能被滥用

**顶级解决方案**（Redis + Token Bucket）：

```python
from datetime import datetime, timedelta
from collections import defaultdict
import time

class RateLimiter:
    """速率限制器（令牌桶算法）"""
    
    def __init__(self, 
                 max_requests: int = 10,  # 最大请求数
                 time_window: int = 60):  # 时间窗口（秒）
        self.max_requests = max_requests
        self.time_window = time_window
        self.requests = defaultdict(list)
    
    def is_allowed(self, user_id: str) -> bool:
        """检查是否允许请求"""
        now = datetime.now()
        
        # 清理过期记录
        self.requests[user_id] = [
            req_time for req_time in self.requests[user_id]
            if now - req_time < timedelta(seconds=self.time_window)
        ]
        
        # 检查是否超限
        if len(self.requests[user_id]) >= self.max_requests:
            return False
        
        # 记录新请求
        self.requests[user_id].append(now)
        return True
    
    def get_retry_after(self, user_id: str) -> int:
        """获取需要等待的秒数"""
        if not self.requests[user_id]:
            return 0
        
        oldest_request = min(self.requests[user_id])
        wait_until = oldest_request + timedelta(seconds=self.time_window)
        wait_seconds = (wait_until - datetime.now()).total_seconds()
        
        return max(0, int(wait_seconds))

# 使用速率限制
rate_limiter = RateLimiter(max_requests=5, time_window=60)

def scan_with_rate_limit(user_id: str):
    """带速率限制的扫描"""
    if not rate_limiter.is_allowed(user_id):
        retry_after = rate_limiter.get_retry_after(user_id)
        st.error(f"⚠️ 请求过于频繁，请{retry_after}秒后重试")
        return None
    
    # 执行扫描
    return scan_stocks()
```

# 使用Redis存储状态
import redis
import pickle

class RedisStateStore:
    """Redis状态存储"""
    
    def __init__(self, redis_url: str):
        self.client = redis.from_url(redis_url)
    
    def set_cache(self, key: str, value: any, ttl: int = 3600):
        """设置缓存"""
        self.client.setex(
            key,
            ttl,
            pickle.dumps(value)
        )
    
    def get_cache(self, key: str) -> any:
        """获取缓存"""
        data = self.client.get(key)
        return pickle.loads(data) if data else None

# 无状态应用
class StatelessAnalyzer:
    """无状态分析器"""
    
    def __init__(self, state_store: RedisStateStore):
        self.state_store = state_store  # 状态存储在外部
    
    def analyze(self, stock_code: str):
        # 从外部存储获取状态
        cache = self.state_store.get_cache(f"stock:{stock_code}")
        if cache:
            return cache
        
        # 计算...
        result = self._compute(stock_code)
        
        # 保存到外部存储
        self.state_store.set_cache(f"stock:{stock_code}", result)
