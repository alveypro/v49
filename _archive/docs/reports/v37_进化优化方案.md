            score += 30

            score += 20

        # 不要太小（容易被操控）也不要太大（难以暴涨）
        if 50亿 < market_cap < 300亿:
            score += 15

        # 它涨的时候，板块其他股票也涨
        if has_leading_effect(stock):
            score += 10
        
        leader_scores.append({
            'stock': stock,
            'leader_score': score,
            'category': categorize_stock(score)
        })
    
    return {
        'leaders': [s for s in leader_scores if s['leader_score'] >= 75],      # 龙头
        'followers': [s for s in leader_scores if 50 <= s['leader_score'] < 75], # 跟随
        'laggards': [s for s in leader_scores if s['leader_score'] < 50]       # 落后
    }
```

def deep_capital_flow_analysis(self, sector, days=5):
    """
    资金流向深度分析（需要level2数据或大单数据）
    """
    # 如果有level2数据
    if has_level2_data():
        return analyze_with_level2()
    
    # 基于普通数据的推算方法
    else:
        stocks = get_sector_stocks(sector)
        capital_metrics = {}
        
        for stock in stocks:
            # 1. 大单推算
            # 涨+高量 = 主力买入
            # 跌+高量 = 主力卖出
            if stock.pct_chg > 0 and stock.vol_ratio > 1.5:
                inflow = stock.amount * 0.7  # 估算70%是买入
            elif stock.pct_chg < 0 and stock.vol_ratio > 1.5:
                inflow = -stock.amount * 0.7  # 流出
            else:
                inflow = 0
            
            # 2. 分时资金推算
            # 尾盘放量+上涨 = 主力资金
            if is_end_of_day_surge(stock):
                inflow *= 1.5  # 主力资金权重更高
            
            # 3. 持续性分析
            # 连续流入3天以上 = 确认资金进场
            consecutive_inflow = count_consecutive_inflow_days(stock)
            
            capital_metrics[stock] = {
                'net_inflow': inflow,
                'consecutive_days': consecutive_inflow,
                'main_capital': inflow if consecutive_inflow >= 3 else 0
            }
        
        # 板块级别汇总
        sector_net_inflow = sum([m['net_inflow'] for m in capital_metrics.values()])
        main_capital_stocks = [s for s, m in capital_metrics.items() 
                               if m['main_capital'] > 0]
        
        # 资金流向分级
        if sector_net_inflow > 10亿 and len(main_capital_stocks) >= 5:
            status = "主力大举进场（强势！）"

            score = 75
        elif sector_net_inflow > 0:
            status = "资金小幅流入（观察）"
            score = 60

            status = "资金撤离（风险！）"
            score = 20

            status = "资金中性"

        return {

            'main_capital_stocks': main_capital_stocks,

        if sentiment > 80:  # 情绪过热
            stop_profit = max(profit_pct - 5, 10)  # 降低目标，准备撤退
            action = "板块过热，建议止盈80%仓位"
        elif profit_pct > 30:
            stop_profit = 30
            action = "已翻倍区域，止盈50%保护利润"
        elif profit_pct > 15:
            stop_profit = 15
            action = "达到第一目标，止盈30%"
        else:
            stop_profit = None
            action = "继续持有"
    
    elif sector_stage == '启动期':
        # 刚启动，不要急着止盈
        stop_profit = 50  # 目标远大
        action = "板块刚启动，持有待涨"
    
    elif sector_stage == '衰退期':
        # 无论盈亏，快速撤离
        stop_profit = max(profit_pct, 0)
        action = "板块衰退，立即清仓"
    
    # 止损策略
    if profit_pct < -5 and sector_sentiment < 40:
        stop_loss = -5
        action = "板块转弱+亏损，止损"
    elif profit_pct < -8:
        stop_loss = -8
        action = "亏损扩大，强制止损"
    else:
        stop_loss = None
    
    # 移动止损（保护利润）
    if profit_pct > 20:
        trailing_stop = profit_pct - 10  # 回调10%就卖
        action = "设置移动止损，保护利润"
    
    return {
        'current_profit': profit_pct,
        'sector_stage': sector_stage,
        'sentiment': sentiment,
        'stop_profit_target': stop_profit,
        'stop_loss_level': stop_loss,
        'action': action
    }
```

**效果**：
- 避免坐电梯（赚钱不卖，回调后变亏损）
- 保护利润（及时止盈）
- 控制风险（严格止损）
- 预期：收益率+30%，风险-50%
